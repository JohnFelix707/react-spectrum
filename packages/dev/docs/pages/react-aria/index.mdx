import { Switch } from "react-aria-components";
import { Finger } from './home/components';

function SwitchExample() {
  let ref = React.useRef();
  let [isAnimating, setAnimating] = React.useState(false);
  let [isSelected, setSelected] = React.useState(true);

  React.useEffect(() => {
    let element = ref.current;
    let observer = new IntersectionObserver((entries) => {
      if (entries[0].isIntersecting) {
        setAnimating(true);
      } else {
        setAnimating(false);
        setSelected(true);
      }
    }, {threshold: 1});
    observer.observe(element);
    return () => observer.unobserve(element);
  }, []);

  return (
    <>
      <Finger style={{animation: isAnimating ? 'touch-animation 12s ease-in-out 500ms' : null}} />
      <Switch
        aria-label="Example switch"
        ref={ref}
        isSelected={isSelected}
        onChange={setSelected}
        className="group inline-flex touch-none"
        style={{ WebkitTapHighlightColor: "transparent" }}>
        <span className="[--bg:theme(colors.slate.300)] dark:[--bg:theme(colors.zinc.600)] bg-[--bg] [--bg-selected:theme(colors.green.500)] group-selected:bg-[--bg-selected] group-focus-visible:outline outline-2 outline-blue-600 mr-4 h-10 w-16 rounded-full border-4 border-transparent outline-offset-2 transition duration-200" style={{animation: isAnimating ? 'switch-background-animation 12s ease-in-out 500ms' : null}}>

    </>

}
