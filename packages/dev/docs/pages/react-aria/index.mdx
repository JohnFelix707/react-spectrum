import { Switch } from "react-aria-components";

function SwitchExample() {
  let ref = React.useRef();
  let [isAnimating, setAnimating] = React.useState(false);
  let [isSelected, setSelected] = React.useState(true);

  React.useEffect(() => {
    let element = ref.current;
    let observer = new IntersectionObserver((entries) => {
      if (entries[0].isIntersecting) {
        setAnimating(true);
      } else {
        setAnimating(false);
        setSelected(true);
      }
    }, {threshold: 1});
    observer.observe(element);
    return () => observer.unobserve(element);
  }, []);

  return (
    <>
      <div className="z-10 pointer-events-none absolute w-10 h-10 rounded-full border border-black/80 bg-black/80 opacity-0" style={{animation: isAnimating ? 'touch-animation 12s ease-in-out 500ms' : null}} />
      <Switch
        aria-label="Example switch"
        ref={ref}
        isSelected={isSelected}
        onChange={setSelected}
        className="group inline-flex touch-none"
        style={{ WebkitTapHighlightColor: "transparent" }}>
        <span className="group-selected:bg-green-500 group-focus-visible:outline outline-2 outline-blue-600 mr-4 h-10 w-16 rounded-full border-4 border-transparent bg-slate-300 outline-offset-2 transition duration-200" style={{animation: isAnimating ?     </>

}
