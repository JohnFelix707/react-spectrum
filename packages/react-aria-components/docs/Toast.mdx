## Example

```tsx example
import {ToastRegion, Toast, ToastQueue, ToastContent, Button, Text} from 'react-aria-components';

const queue = new ToastQueue<any>({
  maxVisibleToasts: 5
});

function Example() {
  return (
    <>
      <ToastRegion toastQueue={queue}>
        {({toast}) => (
          <Toast toast={toast}>
            <ToastContent>
              <Text slot="title">{toast.content.title}</Text>
              <Text slot="description">{toast.content.description}</Text>
            </ToastContent>
            <Button slot="close">x</Button>
          </Toast>
        )}
      </ToastRegion>
      <Button onPress={() => queue.add({title: 'Toast', description: 'Hi'})}>Toast</Button>
    </>
  );
}
```
